```go
// internal/agent/orchestrator.go
func NewOrchestratorAgent(
    ragAgent *agent.Agent,
    summaryAgent *agent.Agent,
) *agent.Agent {
    orch := agent.NewAgent("Orchestrator")

    orch.SetInstructions(`You are AskDoc Assistant.
Your role is to help users find answers from documentation.

Analyze the user's question and decide which specialist agent to handoff:

1. **RAGSearchAgent**: For specific technical questions
2. **SummaryAgent**: For overview questions

Always be helpful and cite sources.`)

    orch.AddHandoff(agent.NewHandoff(ragAgent,
        agent.WithHandoffToolName("search_documents"),
        agent.WithHandoffToolDescription("Search documentation for specific information"),
    ))

    orch.AddHandoff(agent.NewHandoff(summaryAgent,
        agent.WithHandoffToolName("summarize_topic"),
        agent.WithHandoffToolDescription("Summarize a topic or concept"),
    ))

    return orch
}
```
